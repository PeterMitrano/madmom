def plot_features(i):
    plt.imshow(x[i].T)
    plt.title("Plot of features for {}".format(sample_names[i]))
    plt.tight_layout()
    plt.ylabel("features")
    plt.xlabel("time (frame idx)")
    plt.gca().get_yaxis().set_visible(False)    
    plt.show()
    
def plot_responses(i):
    plt.plot(labels[i])
    plt.show()
